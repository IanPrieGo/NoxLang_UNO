Primero checamos si es una Suma o Resta, para ello;
    Primero asumimos el el primero token es una Division o Multiplicacion, y checamos las reglas de la division;
        Primero checamos si el Token siguiente del actual, es un signo de Multiplicacion o Division;
        Si no lo es, entonces asumimos que es un numero normal;
        Si si lo es, entonces avanzamos dos 2 tokens (para llegar al ultimo numero de la Div-Mul), 
        y ahora checamos si el suguiente Token es un operador de Division o Multiplicacion;
            Si no lo es, regresamos a la evalucion de la Suma Resta;
            Si si lo es, seguimos checando los tokens hasta que dejen de haber diviiones o multiplicaciones;
            En el momento que estas dejen de aparecer, regresamos a la evalucion de la Suma Resta;
    Ya que tenemos el Primero Operando de la Suma Resta, 
    ahora checamos el si el siguiente token es un operador de Suma o Resta;
    Si no lo es, aqui terminamos;
    Si si lo es, seguimos haciendo la evaluacion que hicimos para el primer operador, hasta que dejen de aparecer operador de Suma o Resta.

sumaResta(){
   let expresion =  divMult();

    while(match(nextToken(), [TokenType.PLUS, TokenType.MINUS])){

        let operator = nextToken();
        advanceIndex(2);
        let secondOperand = divMult();
        
        expresion = new Exp.Bin(expresion, operator, secondOperand)
    }

    return expresion;

}

divMult(){

    let expresion = Exp.Primary(currentToken());

    while(match(nextToken(), [TokenType.MULTIPLY, TokenType.DIVIDE])){

        let operator = nextToken();
        advanceIndex(2);
        let secondOperand = divMult();
        
        expresion = new Exp.Bin(expresion, operator, secondOperand)
    }

    return expresion;

}

match(suspect, typesToCheck){
    for (type of typesToCheck){
        if(suspect.type == type)
        return true;
    }
    return false;
}

Binary{
    constructor(firstOperand, operator, secondOperand){
        this.firstOperand = firstOperand;
        this.operator = operator;
        this.secondOperand = secondOperand;
    }

    toString(){
        return " (" + firstOperand.toString() + " " + operator.toString() + " " + secondOperand.toString() + ") ";
    }
}

Primary{
    constructor(token){
        this.token = token;
    }

    toString(){
        this.token.toString();
    }
}
